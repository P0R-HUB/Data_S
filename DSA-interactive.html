ลำดับขั้นตอนที่ แปลง Input → Output (Input → [Algorithm: step-by-step] → Output)

Lesson 2 — Stack & Queue

โครงสร้าง 2 แบบที่ต่างกันแค่ "เอาของออกตรงไหน"

Stack = LIFO (เข้าทีหลัง ออกก่อน)

เหมือนกองจานซ้อนกัน — วางจานบนสุด หยิบก็หยิบบนสุด operation หลัก: push (ใส่), pop (เอาออก), peek (ดูตัวบน)

กด push เพื่อใส่ค่า — ตัวบนสุดคือ TOP จะถูก pop ออกก่อนเสมอ
Java — MyStack
public class MyStack {
    private int top = -1;                       // ตัวชี้ตำแหน่งบนสุด เริ่ม -1 = ยังว่าง
    private int[] list;                          // array เก็บข้อมูลจริง
    public MyStack(int size) {                   // constructor: กำหนดขนาด stack
        list = new int[size];                    // จอง array ตามขนาดที่ขอ
    }
    public void push(int x) {                    // ใส่ค่า x ลง stack
        top = top + 1;                           // เลื่อน top ขึ้น 1
        list[top] = x;                           // วาง x ไว้ตำแหน่งบนสุด
    }
    public int pop() {                           // เอาค่าบนสุดออก
        int x = list[top];                       // อ่านค่าตำแหน่ง top เก็บไว้
        top = top - 1;                           // เลื่อน top ลง 1 (ลบออกจาก stack)
        return x;                                // คืนค่าที่เอาออก
    }
}

Queue = FIFO (เข้าก่อน ออกก่อน)

เหมือนคนต่อแถว — ใครมาก่อนได้ก่อน operation: enQueue (ต่อท้าย rear), deQueue (เอาออกหน้า front)

FRONT = ตัวที่จะออกก่อน, REAR = ตัวที่เพิ่งเข้า

ปัญหา Queue แบบ Array → Circular Array

ถ้า deQueue แล้วเลื่อนทุกตัวมาชิดซ้ายจะเปลือง (O(n) ต่อครั้ง) วิธีฉลาดคือใช้ Circular Array: เลื่อนแค่ตัวชี้ และวนกลับต้น array ด้วยสูตร

index ถัดไป = (rear + 1) % SIZE  →  พอ rear ถึงท้าย array ก็วนกลับช่อง 0
Circular Queue (SIZE = 8):
สังเกตว่า REAR วนกลับช่อง 0 ได้เมื่อถึงท้าย

การประยุกต์: ตรวจวงเล็บสมดุลด้วย Stack

เจอ ( → push, เจอ ) → pop · จบแล้ว stack ว่าง = สมดุล

Lesson 3 — Linked List ⭐

โครงสร้างข้อมูลแบบลูกโซ่ — แก้จุดอ่อนของ Array เรื่องการเพิ่ม/ลบและการขยายขนาด

แนวคิด: node + pointer

Linked List ไม่เก็บข้อมูลในช่องที่ติดกันเหมือน Array แต่เก็บแต่ละข้อมูลไว้ใน node โดยแต่ละ node มี 2 ส่วน:

  • element — ตัวข้อมูลจริง (เช่น เลข 34)
  • next — pointer (ตัวชี้) บอกว่า node ถัดไปอยู่ที่ไหนในหน่วยความจำ

node สามารถกระจายอยู่ที่ไหนก็ได้ในหน่วยความจำ (non-contiguous) แค่ "ชี้" หากันด้วย next ก็เรียงเป็นลำดับได้ node ตัวสุดท้ายมี next = null (ไม่มีตัวถัดไป)

🧒 อธิบายแบบง่ายสุดๆ
คิดว่า Linked List = ขบวนรถไฟ 🚂🚃🚃🚃
• แต่ละ node = ตู้รถไฟ 1 ตู้ · ข้างในตู้มี 2 อย่าง: ของที่ขนอยู่ (element) กับ ขอเกี่ยวที่ต่อไปตู้ถัดไป (next)
• head = หัวรถจักร (บอกว่าขบวนเริ่มที่ตู้ไหน)
• ตู้สุดท้ายไม่มีตู้ต่อ → ขอเกี่ยวห้อยลม = null
✨ จุดเด็ด: ถ้าอยากเพิ่ม/แทรก/ลบตู้ ไม่ต้องยกทั้งขบวน แค่ปลดขอเกี่ยวแล้วเกี่ยวใหม่ ก็พอ! (ต่างจาก Array ที่เหมือนเก้าอี้ติดพื้นเป็นแถว จะแทรกต้องเลื่อนทุกตัว)

ตำแหน่งในหน่วยความจำติดกัน (contiguous)กระจายที่ไหนก็ได้ เชื่อมด้วย pointer

โค้ด Node และ LinkedList

Java
public class Node {                  // 1 node ของ linked list
    int element;                     // ส่วนเก็บข้อมูล (ค่าจริง)
    Node next;                       // ตัวชี้ไป node ถัดไป (ตัวสุดท้าย = null)
    public Node(int data) {          // constructor: สร้าง node พร้อมใส่ค่า
        this.element = data;         // เก็บค่าที่รับมาไว้ใน element
    }
}

public class LinkedList {
    Node head;                       // ตัวชี้ "หัวขบวน" (node แรกของ list)
    public LinkedList(int element) { // constructor: สร้าง list พร้อม node แรก
        head = new Node(element);    // สร้าง node แรก
        head.next = null;            // ยังไม่มีตัวถัดไป → null
    }

    // เพิ่มที่หัว: สร้าง node ใหม่ ให้ next ชี้ head เดิม แล้วเลื่อน head มาตัวใหม่
    public void addFirst(int element) {
        Node temp = new Node(element); // สร้าง node ใหม่
        temp.next = head;              // ให้ตัวใหม่ชี้ไปหัวเดิม
        head = temp;                   // เลื่อน head มาที่ตัวใหม่ (ตัวใหม่กลายเป็นหัว)
    }

    // ลบตัวหัว: เลื่อน head ไปตัวถัดไป (ตัวเก่าไม่มีใครชี้ = กลายเป็น garbage)
    public void removeFirst() {
        head = head.next;             // head ข้ามไปตัวที่ 2 → ตัวแรกหลุดออกจาก list
    }
}

ลองเล่น: เพิ่ม/ลบ node แบบเห็นภาพ

addFirst เพิ่มที่หัว · removeFirst ลบที่หัว — สังเกต pointer ขยับ ไม่ต้องเลื่อนข้อมูลเหมือน Array

🎬 อนิเมชัน: แทรก / ลบ node กลางขบวน (เหมือนสไลด์)

นี่คือหัวใจของ Linked List — การ "ปลดขอเกี่ยวแล้วเกี่ยวใหม่" กดดูทีละสเตปได้เลย

🧒 อ่านอนิเมชันยังไง: ลูกศร = ขอเกี่ยวที่บอกว่า "ตู้นี้ต่อไปตู้ไหน" · เวลาแทรก/ลบ เราแค่เปลี่ยนปลายลูกศร ไม่ได้ขยับตู้อื่นเลย · ลูกศรสีแดง = อันที่กำลังเปลี่ยนในสเตปนั้น
📝 เคล็ดลับสอบ (อาจารย์บอกเอง): ถ้าไม่มั่นใจจะเขียนโค้ด วาด diagram แบบนี้อธิบายแทนได้ — วาดตู้ (node) → ลากลูกศร next → เขียนกำกับว่าสเตปไหนเปลี่ยนลูกศรเส้นไหน ก็ได้คะแนนความเข้าใจ · สิ่งสำคัญคือ ลำดับการต่อลูกศรต้องถูก (ดูในอนิเมชันด้านบน): ตอนแทรกต้อง "เกี่ยวตัวใหม่ไปข้างหน้าก่อน แล้วค่อยเปลี่ยนตัวข้างหน้ามาชี้ตัวใหม่" — ถ้าสลับลำดับ ขบวนจะขาด!

⚠️ กับดักข้อสอบ: Object Reference (== vs ค่า)

ใน Java ตัวแปรแบบ object (เช่น Integer, String, node) เก็บ reference (ที่อยู่) ไม่ใช่ค่า — ส่วน primitive (int) เก็บค่าจริง

Java — output คืออะไร?
Integer y = new Integer(20);                    // สร้าง object Integer ตัวที่ 1 (ค่า 20)
Integer w = new Integer(20);                    // สร้าง object ตัวที่ 2 (ค่า 20 เหมือนกัน แต่คนละที่อยู่)
if (w == y) System.out.println("1. w == y");   // ❌ ไม่พิมพ์ — == เทียบที่อยู่ ไม่ใช่ค่า
w = y;                                          // ให้ w ชี้ไป object เดียวกับ y
if (w == y) System.out.println("2. w == y");   // ✅ พิมพ์ — ตอนนี้ที่อยู่เดียวกันแล้ว
ผลลัพธ์: พิมพ์แค่ "2. w == y" — เพราะ == เทียบว่า "ชี้ object เดียวกันไหม" ไม่ได้เทียบค่า · new สร้าง object ใหม่เสมอ ได้ที่อยู่ใหม่ · ถ้าจะเทียบค่าต้องใช้ .equals()

Lesson 4 — Linked List (ต่อ) ⭐

พัฒนา Linked List พื้นฐานให้เก่งขึ้น 3 แบบ

ทบทวน: เดินทั่ว list ด้วย while loop

เทคนิคพื้นฐานที่ใช้ซ้ำในเกือบทุก method (printAll, getSize, getLast): ตั้ง current = head แล้ววนจน current == null

Java
public void printAll() {                         // พิมพ์ทุกค่าใน list
    Node current = head;                         // เริ่มที่หัว
    while (current != null) {                     // วนจนกว่าจะหมด list
        System.out.print(current.element + " ");  // พิมพ์ค่าตัวปัจจุบัน
        current = current.next;                   // ขยับไป node ถัดไป
    }
}
public int countNodes() {                        // นับจำนวน node (getSize)
    Node current = head;                         // เริ่มที่หัว
    int count = 0;                               // ตัวนับ
    while (current != null) {                     // เดินทั้ง list
        count++;                                 // เจอ 1 ตัว นับเพิ่ม
        current = current.next;                   // ไปตัวถัดไป
    }
    return count;                                // คืนจำนวนทั้งหมด
}
public int getLast() {                           // หาค่าตัวสุดท้าย
    Node current = head;                         // เริ่มที่หัว
    while (current.next != null)                  // หยุดเมื่อตัวถัดไปเป็น null (= ตัวสุดท้าย)
        current = current.next;                   // เดินต่อไปเรื่อยๆ
    return current.element;                      // คืนค่าตัวสุดท้าย
}

1) Tailed Linked List

ปัญหา: addLast ของ list ธรรมดาต้องเดินไปท้ายสุดทุกครั้ง (O(n)) · แก้โดยเพิ่มตัวชี้ tail เก็บตำแหน่งท้ายไว้ → addLast เหลือ O(1)

ข้อแลกเปลี่ยน: มี pointer เพิ่ม = ต้อง maintain เพิ่ม (ทุกครั้งที่เพิ่ม/ลบท้าย ต้องอัปเดต tail ด้วย)
Java
Node tail;                           // ตัวชี้ "ท้ายขบวน" (เก็บไว้เพื่อไม่ต้องเดินหา)
public void addLast(int element) {   // เพิ่มที่ท้าย — เร็ว O(1) เพราะรู้ตำแหน่งท้ายอยู่แล้ว
    Node temp = new Node(element);   // สร้าง node ใหม่
    tail.next = temp;                // ต่อ node ใหม่ท้ายตัวเดิม
    tail = temp;                     // เลื่อน tail มาที่ตัวใหม่ (ตัวใหม่กลายเป็นท้าย)
}

2) Circular Linked List

ให้ node ท้าย (tail) ชี้กลับไปที่ head → วนซ้ำได้ไม่รู้จบ เหมาะกับงานที่ต้องวนเป็นรอบ (เช่น สลับเทิร์นผู้เล่น)

3) Doubly Linked List

ปกติเดินได้ทางเดียว (ไปหน้า) · Doubly เพิ่มตัวชี้ prev ให้เดินถอยหลังได้ด้วย ต้องสร้าง node แบบใหม่ที่มีทั้ง next และ prev

Java — Doubly Node
public class Node {                  // node ของ Doubly Linked List
    int element;                     // ข้อมูล
    Node next;                       // ตัวชี้ไปข้างหน้า (ตัวถัดไป)
    Node prev;                       // ตัวชี้ถอยหลัง (ตัวก่อนหน้า) ← ของใหม่
    public Node(int data) {          // constructor
        this.element = data;         // เก็บค่า
    }
}

Lesson 5 — ADT & Generics ⭐

มองโครงสร้างข้อมูลเป็น "กล่องดำ" และทำให้เก็บข้อมูลได้ทุกชนิด

Abstract Data Type (ADT)

ADT = ชนิดข้อมูลที่นิยามด้วย "ชุดค่า + ชุด operation" โดยซ่อนวิธี implement ไว้ภายใน ผู้ใช้แค่รู้ว่ามันทำอะไรได้ (push/pop) ไม่ต้องรู้ว่าข้างในใช้ array หรือ linked list

คิดเป็น "black box": รู้ปุ่มที่กดได้ (operation) และผลที่ได้ พอ — ไม่ต้องเปิดดูข้างใน นี่คือความหมายของคำว่า Abstract

Generics <E> — เก็บได้ทุกชนิด

ปัญหา: MyStack เดิมเก็บได้แค่ int · Generics ทำให้ระบุชนิดข้อมูลเป็น "พารามิเตอร์" ได้ → class เดียวใช้กับทุกชนิด

<E> คือ "ช่องว่าง" สำหรับใส่ชนิดข้อมูล — E เป็นเพียงตัวแทน (placeholder) เวลาใช้งานจริงเราแทน E ด้วยชนิดที่ต้องการ เช่นเขียน Vector<Integer> = "E กลายเป็น Integer" · รูปแบบทั่วไป: class ชื่อคลาส<E> { ... }

ตัวอย่าง: กล่อง 3 ใบ คนละชนิด

public class MyStack<E> {            // <E> = stack นี้เก็บชนิดอะไรก็ได้
    private int top = -1;                // ตัวชี้บนสุด เริ่ม -1 (ว่าง)
    public E[] list;                     // array เก็บข้อมูลชนิด E
    public MyStack(int size) {           // constructor
        list = (E[]) new Object[size];   // ⚠️ ต้องสร้างเป็น Object[] แล้ว cast เป็น E[]
    }
    public void push(E x) {              // ใส่ค่าชนิด E
        top++;                           // เลื่อน top ขึ้น
        list[top] = x;                   // วางค่าบนสุด
    }
    public E pop() {                     // เอาค่าบนสุดออก (คืนชนิด E)
        E tmp = list[top];               // อ่านค่าบนสุด
        top--;                           // ลด top
        return tmp;                      // คืนค่า
    }
}

บทที่สำคัญที่สุด — วิธีวัดว่า "โค้ดเรามีประสิทธิภาพแค่ไหน" โดยไม่ขึ้นกับเครื่อง

}

Lesson 7 — Recursion ⭐

ฟังก์ชันที่เรียกตัวเอง เพื่อแก้ปัญหาเวอร์ชันที่เล็กลงเรื่อยๆ

3 องค์ประกอบที่ขาดไม่ได้

  1. Base Case — เงื่อนไขหยุด (ถ้าไม่มี → วนไม่จบ → stack overflow)
  2. Recursive Case — เรียกตัวเองด้วย input ที่เล็กลง
  3. Progress — แต่ละครั้งต้องขยับเข้าใกล้ base case

ตัวอย่าง: factorial — ดู call stack ทำงาน

n! = n × (n−1) × … × 1 · กดดูว่าฟังก์ชันเรียกตัวเองลงไปจนถึง base case แล้วค่อย "คลายค่า" กลับขึ้นมา

🔵 ฟ้า = กำลังเรียกลงไป · 🟠 ส้ม = ถึง base case · 🟢 เขียว = คืนค่ากลับขึ้นมา
Java
public static int factorial(int n) {     // หา n! = n × (n-1) × ... × 1
    if (n == 0 || n == 1) return 1;       // Base case: หยุดเมื่อถึง 0 หรือ 1 (คืน 1)
    return n * factorial(n - 1);          // Recursive case: n คูณผลของ factorial ตัวเล็กกว่า
}

Fibonacci — ทำไมช้า O(2ⁿ)

F(n) = F(n−1) + F(n−2) · เรียกตัวเอง 2 ครั้ง ทุกชั้น → จำนวนการเรียกโตเป็น 2 เท่าทุกชั้น (binary tree) และคำนวณค่าซ้ำเยอะมาก

เปรียบเทียบ recursion ธรรมดา vs แบบมีความจำ (memoization) — เกริ่นนำสู่ Lesson 8

อ่าน recursion ยังไง

"บนลงล่าง เริ่ม main ก่อน" → เรียกลงลึกจนถึง base case แล้วค่อยคลายค่ากลับ (unrolling) ทีละชั้น เช่น sum(5,10) = 10+9+8+7+6+5 = 45

Tower of Hanoi 🗼

ปริศนาคลาสสิก — ตัวอย่างที่สวยงามของ recursion และ O(2ⁿ)

กติกา

  1. ย้ายได้ทีละ 1 แผ่น
  2. ย้ายได้เฉพาะแผ่นบนสุดของเสา
  3. ห้ามวางแผ่นใหญ่ทับแผ่นเล็ก

หัวใจ (Recursive Insight)

ย้าย N แผ่นจาก A → C:

  1. ย้าย N−1 แผ่นบน A → B (ใช้ C เป็นตัวช่วย)
  2. ย้ายแผ่นใหญ่สุด A → C
  3. ย้าย N−1 แผ่น B → C (ใช้ A เป็นตัวช่วย)
ย้ายแล้ว 0 / 7 ครั้ง
N แผ่นใช้ 2ᴺ − 1 ครั้งเสมอ (3 แผ่น = 7, 4 = 15, 10 = 1023)
Java
// n=จำนวนแผ่น, from=เสาต้นทาง, temp=เสาตัวช่วย, to=เสาปลายทาง
public static void solve(int n, char from, char temp, char to) {
    if (n == 1) {                                 // Base case: เหลือแผ่นเดียว
        System.out.println(from + " → " + to);    //   ย้ายตรงไปปลายทางเลย
        return;
    }
    solve(n - 1, from, to, temp);                 // ขั้น 1: ย้าย n-1 แผ่นบนไปพักที่ temp
    System.out.println(from + " → " + to);        // ขั้น 2: ย้ายแผ่นใหญ่สุดไปปลายทาง
    solve(n - 1, temp, from, to);                 // ขั้น 3: ย้าย n-1 แผ่นจาก temp ไปปลายทาง
}

public static void mergeSort(int[] a, int left, int right) {
    if (left >= right) return;                 // base case: เหลือ 1 ตัว (เรียงเสร็จแล้ว)
    int mid = (left + right) / 2;               // หาจุดกึ่งกลาง
    mergeSort(a, left, mid);                    // แบ่งครึ่งซ้าย (recursion)
    mergeSort(a, mid + 1, right);               // แบ่งครึ่งขวา (recursion)
    merge(a, left, mid, right);                 // รวม 2 ครึ่งที่เรียงแล้วเข้าด้วยกัน
}

// รวม a[left..mid] กับ a[mid+1..right] (สองครึ่งที่เรียงแล้ว) ให้เป็นกองเรียงเดียว
public static void merge(int[] a, int left, int mid, int right) {
    int[] tmp = new int[right - left + 1];      // กองใหม่ไว้พักผลลัพธ์
    int i = left, j = mid + 1, k = 0;           // i=หัวกองซ้าย, j=หัวกองขวา
    while (i <= mid && j <= right) {              // ตราบใดที่ทั้งสองกองยังมีของ
        if (a[i] <= a[j]) tmp[k++] = a[i++];     // หยิบตัวน้อยกว่าจากกองซ้าย
        else              tmp[k++] = a[j++];     // หรือจากกองขวา
    }
    while (i <= mid)   tmp[k++] = a[i++];        // เทของที่เหลือกองซ้าย
    while (j <= right) tmp[k++] = a[j++];        // เทของที่เหลือกองขวา
    for (int x = 0; x < tmp.length; x++)         // ก๊อปกลับเข้า array เดิม
        a[left + x] = tmp[x];
}

    public static int removeAt(int[] arr, int size, int index) {
        if (index < 0 || index >= size) return size;  // กันค่าผิด (index นอกขอบเขต)
        for (int i = index; i < size - 1; i++) {      // วนตั้งแต่จุดที่ลบ ไปจนเกือบท้าย
            arr[i] = arr[i + 1];                       // ดึงตัวขวามาทับ (เลื่อนซ้าย 1 ช่อง)
        }
        arr[size - 1] = 0;                             // เคลียร์ช่องท้ายสุดที่ซ้ำ
        return size - 1;                               // คืนขนาดใหม่ (ลดลง 1)
    }

    // (2) คัดลอกทุกสมาชิกจาก src ไป dest
    public static void copyArray(int[] src, int[] dest) {
        for (int i = 0; i < src.length; i++) {        // วนทุก index
            dest[i] = src[i];                          // ก๊อปค่าทีละช่อง
        }
    }

        size = removeAt(a, size, 2);                   // ลบ 30 (index 2) → size เหลือ 4

        int[] b = new int[a.length];                   // สร้างกล่องปลายทาง

โจทย์ในสไลด์: จะเกิดอะไรเมื่อ pop stack ว่าง? เมื่อ stack เต็ม? และจะดูค่า top อย่างไร?

public class MyStack {
    private int top = -1;                     // ตัวชี้บนสุด (-1 = ว่าง)
    private int[] list;                        // array เก็บข้อมูล

    public MyStack(int size) {                 // constructor กำหนดขนาด
        list = new int[size];
    }

    public boolean isEmpty() { return top == -1; }              // ว่างไหม
    public boolean isFull()  { return top == list.length - 1; } // เต็มไหม
    public int size()        { return top + 1; }                // มีกี่ตัว

    public void push(int x) {                  // ใส่ค่า (เวอร์ชันกันพลาด)
        if (isFull()) {                        // ถ้าเต็มแล้ว
            System.out.println("Stack เต็ม! push(" + x + ") ไม่ได้");
            return;                            // ออกเลย ไม่ใส่
        }
        top = top + 1;                         // เลื่อน top ขึ้น
        list[top] = x;                         // วางค่า
    }

    public int pop() {                         // เอาออก (เวอร์ชันกันพลาด)
        if (isEmpty()) {                       // ถ้าว่าง
            System.out.println("Stack ว่าง! pop ไม่ได้");
            return -1;                         // คืน -1 เป็นสัญญาณว่าผิดพลาด
        }
        int x = list[top];                     // อ่านค่าบนสุด
        top = top - 1;                         // ลด top
        return x;                              // คืนค่า
    }

    public int peek() {                        // ดูค่า top โดยไม่เอาออก
        if (isEmpty()) return -1;              // ว่าง → -1
        return list[top];                      // คืนค่าบนสุด (ไม่ลด top)
    }

    public static void main(String[] args) {
        MyStack s = new MyStack(3);            // stack จุ 3
        s.push(5); s.push(3); s.push(4);       // ใส่ 5,3,4 (เต็มพอดี)
        s.push(9);                             // เต็มแล้ว → พิมพ์เตือน ไม่ใส่
        System.out.println("top = " + s.peek());    // 4 (ดูเฉยๆ)
        System.out.println("pop = " + s.pop());      // 4 (เอาออกจริง)
        System.out.println("size = " + s.size());    // 2 (เหลือ 5,3)
    }
}

public class BalancedBrackets {
    public static boolean isBalanced(String expr) {  // คืน true ถ้าวงเล็บสมดุล

        }

    }

class Node {
    int element;                                // ข้อมูล
    Node next;                                  // ตัวชี้ตัวถัดไป
    public Node(int data) { this.element = data; }  // constructor
}

public class LinkedList {
    Node head;                                  // ตัวชี้หัว
    Node tail;                                  // Tailed: เก็บตัวท้ายไว้ด้วย

    public LinkedList(int element) {            // สร้าง list พร้อม node แรก
        head = new Node(element);               // node แรก
        head.next = null;                       // ยังไม่มีตัวถัดไป
        tail = head;                            // มีตัวเดียว หัว=ท้าย
    }

    // เพิ่มที่หัว
    public void addFirst(int element) {
        Node temp = new Node(element);          // สร้างตัวใหม่
        temp.next = head;                       // ให้ชี้หัวเดิม
        head = temp;                            // เลื่อนหัวมาตัวใหม่
    }

    // เพิ่มที่ท้าย — O(1) เพราะมี tail (ไม่ต้องเดินหา)
    public void addLast(int element) {
        Node temp = new Node(element);          // สร้างตัวใหม่
        tail.next = temp;                       // ต่อท้ายตัวเดิม
        tail = temp;                            // เลื่อนท้ายมาตัวใหม่
    }

    // ลบตัวหัว
    public void removeFirst() {
        if (head == null) return;               // ว่างอยู่ ไม่ต้องทำ
        head = head.next;                       // ข้ามไปตัว 2 (ตัวเก่ากลายเป็น garbage)
    }

    // พิมพ์ทั้งหมด — แบบ loop
    public void printAll() {
        Node current = head;                    // เริ่มที่หัว
        while (current != null) {                // วนจนหมด
            System.out.print(current.element + " ");  // พิมพ์ค่า
            current = current.next;              // ไปตัวถัดไป
        }
        System.out.println();                   // ขึ้นบรรทัดใหม่
    }

    // พิมพ์ทั้งหมด — แบบ recursion
    public void printElement(Node x) {
        if (x == null) return;                  // base case: ถึง null = จบ
        System.out.print(x.element + " ");      // พิมพ์ตัวปัจจุบัน
        printElement(x.next);                   // recursive case: เรียกตัวถัดไป
    }

    // นับจำนวน node
    public int getSize() {
        Node current = head;                    // เริ่มหัว
        int count = 0;                          // ตัวนับ
        while (current != null) {                // เดินทั้ง list
            count++;                            // นับเพิ่ม
            current = current.next;             // ไปตัวถัดไป
        }
        return count;                           // คืนจำนวน
    }

    // หาค่าตัวสุดท้าย
    public int getLast() {
        Node current = head;                    // เริ่มหัว
        while (current.next != null)             // หยุดเมื่อตัวถัดไป=null
            current = current.next;             // เดินต่อ
        return current.element;                 // คืนค่าตัวท้าย
    }

    public static void main(String[] args) {
        LinkedList l = new LinkedList(5);
        l.addFirst(8); l.addFirst(9); l.addFirst(2); l.addFirst(6);  // เพิ่มที่หัวทีละตัว
        l.printElement(l.head);                 // พิมพ์แบบ recursion → 6 2 9 8 5
        l.removeFirst();                        // ลบหัว (6) ออก
        l.printAll();                           // พิมพ์แบบ loop → 2 9 8 5
        System.out.println("size = " + l.getSize());  // 4
        System.out.println("last = " + l.getLast());  // 5
    }
}

public class RecursionHomework {

    // พิมพ์ "I Miss You" n ครั้ง (โจทย์ GreetingCard)
    public static void forgetMeNot(int n) {
        if (n != 0) {                           // base case: n == 0 หยุด
            System.out.println("I Miss You");
            forgetMeNot(n - 1);                 // recursive call
        }
    }

    // ผลรวม start..end
    public static int sum(int start, int end) {
        if (end <= start) return end;           // base case: ถึงตัวเริ่ม คืนค่านั้น
        return end + sum(start, end - 1);       // recursive case: end + ผลรวมที่เหลือ
    }

    // Fibonacci
    public static int fib(int num) {
        if (num == 0) return 0;                 // base case ตัวที่ 1
        if (num == 1) return 1;                 // base case ตัวที่ 2
        return fib(num - 1) + fib(num - 2);     // เรียกตัวเอง 2 ครั้ง → O(2^n)
    }

    public static void main(String[] args) {
        forgetMeNot(3);                         // พิมพ์ "I Miss You" 3 ครั้ง

    }

Lesson 8 — DP: Coin Change & Knapsack (bottom-up)

กิจกรรมในสไลด์: coins=[1,2,5] amount=7 → เฉลย 2 เหรียญ (5+2)

Java
import java.util.Arrays;

public class DPHomework {

    // จำนวนเหรียญน้อยสุด (คืน -1 ถ้าจ่ายไม่ได้)
    public static int coinChange(int[] coins, int amount) {
        int[] dp = new int[amount + 1];         // dp[a] = เหรียญน้อยสุดสำหรับยอด a
        Arrays.fill(dp, amount + 1);            // เติม "อนันต์" (amount+1 = มากเกินจริง)
        dp[0] = 0;                              // ยอด 0 ใช้ 0 เหรียญ (base case)
        for (int i = 1; i <= amount; i++)        // ไล่ทุกยอดตั้งแต่ 1 ถึง amount
            for (int coin : coins)               // ลองทุกชนิดเหรียญ
                if (coin <= i)                    // ใช้เหรียญนี้ได้ถ้าไม่เกินยอด
                    dp[i] = Math.min(dp[i], dp[i - coin] + 1);  // เลือกที่น้อยกว่า (ใช้เหรียญนี้ +1)
        return dp[amount] > amount ? -1 : dp[amount];  // ยังเป็นอนันต์ = จ่ายไม่ได้ → -1
    }

    // 0/1 Knapsack — มูลค่าสูงสุด
    public static int knapsack(int[] w, int[] v, int capacity) {
        int n = w.length;                       // จำนวนของ
        int[][] dp = new int[n + 1][capacity + 1];  // dp[i][c]=มูลค่าสูงสุด i ชิ้นแรก จุ c
        for (int i = 1; i <= n; i++)             // ไล่ทีละชิ้น
            for (int c = 1; c <= capacity; c++) {        // ไล่ทุกความจุ
                if (w[i - 1] > c) dp[i][c] = dp[i - 1][c];           // หนักเกิน → skip (ใช้ค่าแถวบน)
                else dp[i][c] = Math.max(dp[i - 1][c],               // เทียบ: ไม่เอาชิ้นนี้ (skip)
                                v[i - 1] + dp[i - 1][c - w[i - 1]]); // กับ: เอาชิ้นนี้ (take)
            }
        return dp[n][capacity];                 // คำตอบอยู่ช่องขวาล่างสุด
    }

    public static void main(String[] args) {
        System.out.println(coinChange(new int[]{1,2,5}, 7));   // 2 (5+2)
        System.out.println(coinChange(new int[]{1,3,4}, 6));   // 2 (3+3)
        System.out.println(knapsack(new int[]{1,3,4,5},
                                    new int[]{1,4,5,7}, 7));   // 9 (ของน้ำหนัก 3+4)
    }
